How often does the home team win

playoffs_recent %>% 
  filter(game_number == max_game_number) %>% 
  mutate(
    home_winner = if_else(home_pts > visitor_pts, 1L, 0L)
  ) %>%
  count(home_winner) %>% 
  mutate(pct = n /sum(n))
